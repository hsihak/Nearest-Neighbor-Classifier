139// nearestNeighbour function that check the nearest neighbour with the train data
140Data& NNClassifier::nearestNeighbour(Data point)
141{
142 Data nearestNeighbour = dataPoints[0];
143
144 for (int i = 1; i < ARRAYSIZE; i++)
145 {
146 if (point.getDistanceFrom(dataPoints[i]) < point.getDistanceFrom(nearestNeighbour))
147 {
148 nearestNeighbour = dataPoints[i];

150 }
151
